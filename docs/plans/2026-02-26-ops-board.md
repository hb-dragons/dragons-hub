# Club Operations Board Implementation Plan

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Add a Kanban board with automated venue booking tracking that responds to match schedule changes from sync and local edits.

**Architecture:** New DB schema for venue bookings, boards, tasks, and notifications. A `VenueBookingService` reconciles bookings after sync and local edits. A `NotificationService` with pluggable providers (in-app first, WhatsApp later). Frontend Kanban board at `/admin/board` and booking overview at `/admin/bookings`.

**Tech Stack:** Drizzle ORM schemas, Hono API routes, Zod validation, Next.js server components + SWR client components, @dnd-kit for drag-and-drop, BullMQ for notification queue.

```typescript
// booking-calculator.ts
export interface BookingMatchInput {
  kickoffTime: string; // "HH:mm:ss"
  teamGameDuration: number | null; // minutes, null = use default
}

export interface BookingConfig {
  bufferBeforeMinutes: number;
  bufferAfterMinutes: number;
  defaultGameDurationMinutes: number;
}

export interface TimeWindow {
  calculatedStartTime: string; // "HH:mm:ss"
  calculatedEndTime: string; // "HH:mm:ss"
}

function timeToMinutes(time: string): number {
  const [h, m] = time.split(":").map(Number);
  return h * 60 + m;
}

function minutesToTime(minutes: number): string {
  const h = Math.floor(minutes / 60);
  const m = minutes % 60;
  return `${String(h).padStart(2, "0")}:${String(m).padStart(2, "0")}:00`;
}

export function calculateTimeWindow(
  matches: BookingMatchInput[],
  config: BookingConfig,
): TimeWindow | null {
  if (matches.length === 0) return null;

  let earliestKickoff = Infinity;
  let latestEnd = -Infinity;

  for (const match of matches) {
    const kickoff = timeToMinutes(match.kickoffTime);
    const duration = match.teamGameDuration ?? config.defaultGameDurationMinutes;
    const end = kickoff + duration;

    if (kickoff < earliestKickoff) earliestKickoff = kickoff;
    if (end > latestEnd) latestEnd = end;
  }

  return {
    calculatedStartTime: minutesToTime(earliestKickoff - config.bufferBeforeMinutes),
    calculatedEndTime: minutesToTime(latestEnd + config.bufferAfterMinutes),
  };
}
```
